#выполняем комманды от суперпользователя (под root)
#создаем группы
groupadd bookkeepers
groupadd storekeepers
groupadd developers

#создаем дирректории
mkdir /var/data
mkdir /var/data/invoices
mkdir /var/data/web
mkdir /var/data/store

#создаем пользователя admin
useradd -s /bin/bash -m -p qwerty admin

#назначаем admin владельцем соответствующих дирректорий
chown -R admin /var/data/invoices
chown -R admin /var/data/web
chown -R admin /var/data/store

#переходим в дирректорию data
cd /var/data

#назначаем права соответствующим дирректориям
chmod 774 invoices
chmod 774 web
chmod 774 store
